       rep.ID,
       rep.REP_CODE      SUBREPORT_CODE,
       rep.REP_NAME      SUBREPORT_NAME,
       rep.REP_TYPE      SUBREPORT_TYPE,
       decode(rep.REP_TYPE, 0, 'Crystal Reports', 1, 'Web-форма (frm)', 2, 'Crystal Reports (PDF)', 3, 'Бланк', 5, 'WEB-конструктор', 6, 'Составной', null) SUBREPORT_STYPE,
       t2.FULLNAME      SUBREPORT_LPU_NAME,
       rep.IS_SHARE      SUBREPORT_IS_SHARE,
       t.SORT,
       rep.REP_TYPE, --Тип (по виду продукта): 0 - Crystal Reports; 1 - WEB-форма; 2 - Crystal Reports(PDF); 3 - Бланк; 5 - WEB-конструктор; 6 - Составной
       rep.REP_DATA,
       rep.REP_FILENAME ,
       rep.REP_NAME,
       rep.REP_CODE,
       drl.PRIV_NAME      -- Добавленное поле
  from D_REPORTS_STRUCTURE             t
  join D_REPORTS                       rep on rep.ID = t.SUBREPORT
  join D_LPU                           t2 on rep.LPU = t2.ID
  join D_REPORTS_LINKS                 drl on drl.pid = rep.id
 where t.PID = 382272424 -- 455125366-- 423419106  -- r.ID
   and exists (select null from D_V_URPRIVS ur where ur.CATALOG = t.CID and ur.UNITCODE = 'REPORTS_STRUCTURE')
